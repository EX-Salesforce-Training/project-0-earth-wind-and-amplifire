public interface Ship {
    
    // Set the length of this ship.
    void setLength(integer length);
    
    // Get the length of this ship.
    integer getLength();
    
    // Set the weight of the ship.
    void setWeight(integer weight);
    
    // Get the weight of the ship.
    integer getWeight();
    
    // Load items onto the ship.
    void load(List<string> cargo);
    
    // Unload items from the ship.
    List<string> unload();
    
    // Get those barnacles off of the ship.
    void removeBarnacles();
    
    // Look for stowaways.
    List<string> locateStowaways();

}
